def create_data(args, T=10, n=10, split=0.8):
    x = np.zeros((n, 2, T))
    y = np.zeros((n,))
    idx_arr = np.arange(T)

    for i in range(n):
        idx = np.random.choice(idx_arr, 2, replace=False)
        x[i, 0, idx] = 1
        x[i, 1, :] = np.random.uniform(0, 1, size=T)
        y[i] = np.dot(x[i, 1], x[i, 0].T)

    x = np.swapaxes(x, 2, 1)

    row = round(n*split)

    return x[:row], y[:row], x[row:], y[row:]
